00168 inline int
00169 pow2roundup (int x)
00170 {
00171     // Here's a version with no loops.
00172     if (x < 0)
00173         return 0;
00174     // Subtract 1, then round up to a power of 2, that way if we are
00175     // already a power of 2, we end up with the same number.
00176     --x;
00177     // Make all bits past the first 1 also be 1, i.e. 0001xxxx -> 00011111
00178     x |= x >> 1;
00179     x |= x >> 2;
00180     x |= x >> 4;
00181     x |= x >> 8;
00182     x |= x >> 16;
00183     // Now we have 2^n-1, by adding 1, we make it a power of 2 again
00184     return x+1;
00185 }

00191 inline int
00192 pow2rounddown (int x)
00193 {
00194     // Make all bits past the first 1 also be 1, i.e. 0001xxxx -> 00011111
00195     x |= x >> 1;

00403 D convert_type (const S &src)
00404 {
00405     if (is_same<S,D>::value) {
00406         // They must be the same type.  Just return it.
00407         return (D)src;

00410     F scale = std::numeric_limits<S>::is_integer ?
00411         ((F)1.0)/std::numeric_limits<S>::max() : (F)1.0;

00413         // Converting to an integer-like type.
00414         typedef double F;
00415         F min = (F) std::numeric_limits<D>::min();
00416         F max = (F) std::numeric_limits<D>::max();
00417         scale *= max;

00419     } else {
00420         // Converting to a float-like type, so we don't need to remap

00720 {
00721     if (f <= 0) {   // Trivial case of zero, and handle all negative values
00722         num = 0;
